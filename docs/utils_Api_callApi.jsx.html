import axios from 'axios';

/**
 * Get data from API on user
 * @param {string} id
 * @returns Object
 */

async function DataUser(id) {
	try {
		const res = await axios.get(`http://localhost:3000/user/${id}`);
		return res.data;
	} catch (e) {
		console.log(e);
	}
}
